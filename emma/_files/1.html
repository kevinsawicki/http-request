272                /**
273                 * <p>
274                 * Encodes up to three bytes of the array <var>source</var> and writes
275                 * the resulting four Base64 bytes to <var>destination</var>. The source
276                 * and destination arrays can be manipulated anywhere along their length
277                 * by specifying <var>srcOffset</var> and <var>destOffset</var>. This
278                 * method does not check to make sure your arrays are large enough to
279                 * accomodate <var>srcOffset</var> + 3 for the <var>source</var> array
280                 * or <var>destOffset</var> + 4 for the <var>destination</var> array.
281                 * The actual number of significant bytes in your array is given by
282                 * <var>numSigBytes</var>.
283                 * </p>
284                 * <p>
285                 * This is the lowest level of the encoding methods with all possible
286                 * parameters.
287                 * </p>
288                 *
289                 * @param source
290                 *            the array to convert
291                 * @param srcOffset
292                 *            the index where conversion begins
293                 * @param numSigBytes
294                 *            the number of significant bytes in your array
295                 * @param destination
296                 *            the array to hold the conversion
297                 * @param destOffset
298                 *            the index where output will be put
299                 * @return the <var>destination</var> array
300                 * @since 1.3
301                 */
302                private static byte[] encode3to4(byte[] source, int srcOffset,
303                                int numSigBytes, byte[] destination, int destOffset) {
304 
305                        byte[] ALPHABET = _STANDARD_ALPHABET;
306 
307                        int inBuff = (numSigBytes > 0 ? ((source[srcOffset] << 24) >>> 8)
308                                        : 0)
309                                        | (numSigBytes > 1 ? ((source[srcOffset + 1] << 24) >>> 16)
310                                                        : 0)
311                                        | (numSigBytes > 2 ? ((source[srcOffset + 2] << 24) >>> 24)
312                                                        : 0);
313 
314                        switch (numSigBytes) {
315                        case 3:
316                                destination[destOffset] = ALPHABET[(inBuff >>> 18)];
317                                destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
318                                destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
319                                destination[destOffset + 3] = ALPHABET[(inBuff) & 0x3f];
320                                return destination;
321 
322                        case 2:
323                                destination[destOffset] = ALPHABET[(inBuff >>> 18)];
324                                destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
325                                destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
326                                destination[destOffset + 3] = EQUALS_SIGN;
327                                return destination;
328 
329                        case 1:
330                                destination[destOffset] = ALPHABET[(inBuff >>> 18)];
331                                destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
332                                destination[destOffset + 2] = EQUALS_SIGN;
333                                destination[destOffset + 3] = EQUALS_SIGN;
334                                return destination;
335 
336                        default:
337                                return destination;
338                        }
339                }

407                 *            Offset in array where conversion should begin
408                 * @param len

414                 *             if source array, offset, or length are invalid
415                 * @since 2.3.1
416                 */
417                public static byte[] encodeBytesToBytes(byte[] source, int off, int len) {
418 
419                        if (source == null)
420                                throw new NullPointerException("Cannot serialize a null array.");
421 
422                        if (off < 0)
423                                throw new IllegalArgumentException(
424                                                "Cannot have negative offset: " + off);
425 
426                        if (len < 0)
427                                throw new IllegalArgumentException(
428                                                "Cannot have length offset: " + len);
429 
430                        if (off + len > source.length)
431                                throw new IllegalArgumentException(
432                                                String.format(
433                                                                "Cannot have offset of %d and length of %d with array of length %d",
434                                                                off, len, source.length));
435 
436                        // Bytes needed for actual encoding
437                        int encLen = (len / 3) * 4 + (len % 3 > 0 ? 4 : 0);
438 
439                        byte[] outBuff = new byte[encLen];
440 
441                        int d = 0;
442                        int e = 0;
443                        int len2 = len - 2;
444                        for (; d < len2; d += 3, e += 4)
445                                encode3to4(source, d + off, 3, outBuff, e);
446 
447                        if (d < len) {
448                                encode3to4(source, d + off, len - d, outBuff, e);
449                                e += 4;
450                        }
451 
452                        if (e <= outBuff.length - 1) {
453                                byte[] finalOut = new byte[e];
454                                System.arraycopy(outBuff, 0, finalOut, 0, e);
455                                return finalOut;
456                        } else
457                                return outBuff;
458                }
